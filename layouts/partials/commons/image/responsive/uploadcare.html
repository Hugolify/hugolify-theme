{{/* Responsive image (srcset width descriptors) — Uploadcare. Transformations are generated on the fly (no build cost), so a dense ladder is fine. Retina is handled by the browser via DPR; variants larger than the source are skipped. */}} {{- $file := .file -}} {{- $quality := site.Params.image.quality | default "normal" -}} {{/* crop alignment (lowercased); crop_ratio (h/w) enables cropped variants via scale_crop. */}} {{- $crop := lower (.crop | default "center") -}} {{- $crop_ratio := .crop_ratio -}} {{/* Width ladder: explicit caller widths, else the dense CDN ladder. */}} {{- $widths := .widths -}} {{- if not $widths -}} {{- $widths = site.Params.image.srcset_widths | default (slice 320 480 640 768 960 1280 1600 1920) -}} {{- end -}} {{/* Strip filename from Uploadcare URL if present (/UUID/file.ext -> /UUID/). */}} {{- $stripFilenamePattern := `(/[a-f0-9-]+)/[^/]+\.[a-zA-Z]+/?$` -}} {{- $file = replaceRE $stripFilenamePattern "$1/" $file -}} {{- $image_desktop := $file -}} {{- $image_width := false -}} {{- $image_height := false -}} {{- $srcset := slice -}} {{- if not (strings.HasSuffix $file "svg") -}} {{/* Intrinsic size: caps the ladder and feeds the aspect ratio. */}} {{- $size := partial "func/GetRemoteImageSize.html" (dict "url" (printf "%s-/json/" $file) "provider" "uploadcare") -}} {{- $image_width = $size.width -}} {{- $image_height = $size.height -}} {{/* format/auto: Uploadcare negotiates AVIF/WebP/original per browser via the Accept header, so a single srcset suffices — no hand-built webp/avif needed. */}} {{- $q := printf "-/format/auto/-/quality/%s/" $quality -}} {{- $fallback := false -}} {{- $fallback_w := false -}} {{- $fallback_h := false -}} {{- range $w := $widths -}} {{- $w = int $w -}} {{- if or (not $image_width) (le $w (int $image_width)) -}} {{- if $crop_ratio -}} {{/* Crop: derive height from the requested ratio, scale_crop to w×h with the chosen alignment. */}} {{- $h := int (math.Round (mul (float $w) $crop_ratio)) -}} {{- $srcset = $srcset | append (printf "%s-/scale_crop/%dx%d/%s/%s %dw" $file $w $h $crop $q $w) -}} {{- $fallback = printf "%s-/scale_crop/%dx%d/%s/%s" $file $w $h $crop $q -}} {{- $fallback_w = $w -}} {{- $fallback_h = $h -}} {{- else -}} {{- $srcset = $srcset | append (printf "%s-/resize/%dx/%s %dw" $file $w $q $w) -}} {{- $fallback = printf "%s-/resize/%dx/%s" $file $w $q -}} {{- end -}} {{- end -}} {{- end -}} {{- $image_desktop = $fallback | default $file -}} {{/* For a crop, the dimensions are the fallback variant's computed crop size. */}} {{- if $crop_ratio -}} {{- $image_width = $fallback_w -}} {{- $image_height = $fallback_h -}} {{- end -}} {{- end -}} {{- return dict "image_desktop" $image_desktop "image_width" $image_width "image_height" $image_height "srcset" $srcset -}}
