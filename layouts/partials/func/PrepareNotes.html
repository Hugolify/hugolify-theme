{{/* Renders markdown text and injects note links in place of [*] markers. The [*] markers are swapped for a neutral placeholder *before* markdownify, then restored *after*. This avoids two pitfalls: 1. The '*' inside [*] being consumed by markdown emphasis parsing (e.g. "note[*]. Et *italique*" would pair the asterisks). 2. Goldmark stripping raw note HTML when injected before markdownify (renderer.unsafe defaults to false). */}} {{ $text := .html }} {{ $notes := .notes }} {{ $blockId := .blockId | default "" }} {{/* Plain-ASCII sentinel: markdown never transforms it, virtually no collision risk */}} {{ $token := "xHugolifyNotex" }} {{/* 1. Protect each [*] with an indexed placeholder before markdownify */}} {{ $protected := $text }} {{ range $i, $_ := findRESubmatch `\[\*\]` $text }} {{ $placeholder := printf "%s%d%s" $token (add $i 1) $token }} {{ $protected = replaceRE `\[\*\]` $placeholder $protected 1 }} {{ end }} {{/* 2. Apply punctuation rules then render markdown */}} {{ $rendered := partial "func/PrepareHTML.html" $protected | markdownify }} {{/* 3. Restore placeholders with the note links */}} {{ range $index, $note := $notes }} {{ $noteIndex := add $index 1 }} {{ $id := print "-" $noteIndex | print $blockId }} {{ $linkId := print "nh-" $id }} {{ $noteId := print "nb-" $id }} {{ $noteContent := $note.text }} {{ $replacement := "" }} {{ if $noteContent }} {{ $replacement = printf `[%d]` $linkId $noteId $noteId $noteContent $noteIndex }} {{ end }} {{ $placeholder := printf "%s%d%s" $token $noteIndex $token }} {{ $rendered = replace $rendered $placeholder $replacement }} {{ end }} {{/* Strip any leftover placeholder with no matching note */}} {{ $rendered = replaceRE (printf "%s\\d+%s" $token $token) "" $rendered }} {{ return safeHTML $rendered }}
