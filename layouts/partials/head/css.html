{{/* CSS — styling module agnostic Auto-detection (in order): sass/main.sass → SASS indented syntax (hugolify-theme-bootstrap) sass/main.scss → SCSS syntax css/main.css → native CSS pipeline (hugolify-theme-design-system) Site params: params.css.postcss: "postcss/design-system" → directory containing postcss.config.js (default: project root) params.css.preload: true → adds params.css.print: true → loads sass/print.sass, sass/print.scss or css/print.css with media="print" */}} {{- $params := .Site.Params -}} {{/* --- Main stylesheet --- */}} {{/* Options */}} {{- $cssOpts := (dict "targetPath" "assets/css/main.css" "enableSourceMap" (not hugo.IsProduction) "includePaths" (slice "node_modules") ) -}} {{/* PostCSS — config directory (default: project root) */}} {{- $postCSSDirectoryConfig := $params.css.postcss | default "." -}} {{/* CSS pipeline */}} {{- $css := "" -}} {{- if resources.Get "sass/main.sass" -}} {{- $css = resources.Get "sass/main.sass" | toCSS $cssOpts -}} {{- else if resources.Get "sass/main.scss" -}} {{- $css = resources.Get "sass/main.scss" | toCSS $cssOpts -}} {{- else if resources.Get "css/main.css" -}} {{- $css = resources.Get "css/main.css" -}} {{- end -}} {{- if $css -}} {{- $css = $css | css.PostCSS (dict "config" $postCSSDirectoryConfig "inlineImports" true "skipInlineImportsNotFound" true) -}} {{- end -}} {{- if and hugo.IsProduction $css -}} {{- $css = $css | minify | fingerprint | resources.PostProcess -}} {{- end -}} {{/* HTML output */}} {{- with $css -}} {{- if $params.css.preload }} {{- end }} {{- end }} {{/* --- Print stylesheet (optional) --- */}} {{- if $params.css.print -}} {{/* Options */}} {{- $printOpts := (dict "targetPath" "assets/css/print.css" "enableSourceMap" (not hugo.IsProduction) "includePaths" (slice "node_modules") ) -}} {{/* CSS pipeline */}} {{- $print := "" -}} {{- if resources.Get "sass/print.sass" -}} {{- $print = resources.Get "sass/print.sass" | toCSS $printOpts -}} {{- else if resources.Get "sass/print.scss" -}} {{- $print = resources.Get "sass/print.scss" | toCSS $printOpts -}} {{- else if resources.Get "css/print.css" -}} {{- $print = resources.Get "css/print.css" -}} {{- end -}} {{- if and hugo.IsProduction $print -}} {{- $print = $print | minify | fingerprint | resources.PostProcess -}} {{- end -}} {{/* HTML output */}} {{- with $print }} {{- end -}} {{- end -}}
